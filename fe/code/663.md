``` js
const random = n => Math.floor(Math.random() * (n + 1))

// 可生成重复的随机验证码
function randomCode () {
  return Array(6).fill().map(() => random(9))
}

function randomCode2 () {
  let i = 0;
  let l = []
  while (i < 6) {
    const x = random(9)
    if (!l.includes(x)) {
      i++;
      l.push(x)
    }
  }
  return l
}
```
