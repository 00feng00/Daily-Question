# 一道异步代码执行输出顺序问题

::: tip 更多描述 
 ``` js
setTimeout(() => {
  console.log('A')
  Promise.resolve().then(() => {
    console.log('B')
  })
}, 1000)

Promise.resolve().then(() => {
  console.log('C')
})

new Promise((resolve) => {
  console.log('D')
  resolve('')
}).then(() => {
  console.log('E')
})

async function sum (a, b) {
  console.log('F')
}

async function asyncSum(a, b) {
  await Promise.resolve()
  console.log('G')
  return Promise.resolve(a + b)
}

sum(3, 4)
asyncSum(3, 4)
console.log('H')
```
